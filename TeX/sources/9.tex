%読み込んだ文字列\Iを反転させ\rIに入れる
\read-1to\I
\def\rI{}
\def\appendreverse#1#2{%
\expandafter\ifx#2;\edef\rI{#1\rI}
\else\edef\rI{#1\rI}\expandafter\expandafter\expandafter\appendreverse\expandafter#2\fi}
\expandafter\appendreverse\I;

%%最初の７文字で場合分け
\def\flag{YES}
\def\trim#1#2#3#4#5#6#7#8-{%
\edef\five{#1#2#3#4#5}
\edef\six{#1#2#3#4#5#6}
\edef\seven{#1#2#3#4#5#6#7}
\def\dream{maerd}
\def\dreamer{remaerd}
\def\erase{esare}
\def\eraser{resare}
\expandafter\ifx#1;\else
\ifx\five\dream\expandafter\expandafter\expandafter\expandafter\expandafter\expandafter\expandafter\trim\expandafter\expandafter\expandafter#6\expandafter#7#8-\else
\ifx\five\erase\expandafter\expandafter\expandafter\expandafter\expandafter\expandafter\expandafter\trim\expandafter\expandafter\expandafter#6\expandafter#7#8-\else
\ifx\six\eraser\expandafter\expandafter\expandafter\trim\expandafter#7#8-\else
\ifx\seven\dreamer\expandafter\trim#8-\else
\def\flag{NO}\fi\fi\fi\fi\fi
}
\expandafter\trim\rI;::::::-
\immediate\write16{\flag}
\end
